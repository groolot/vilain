\newcommand{\image}[5][\textwidth]{%
% #1 = minipage width
% #2 = file path
% #3 = includegraphics options
% #4 = caption text
% #5 = reference label
    \begin{figure}[h]
        \centering
            \begin{minipage}[c]{#1}
                \centering
                \includegraphics[#3]{#2}
                \caption{#4}
                \label{#5}
            \end{minipage}
    \end{figure}
}
\newcommand{\imageBoxed}[5][\textwidth]{%
% #1 = minipage width
% #2 = file path
% #3 = includegraphics options
% #4 = caption text
% #5 = reference label
    \begin{figure}[h]
        \centering
        \fbox{
            \begin{minipage}[c]{#1}
                \centering
                \includegraphics[#3]{#2}
                \caption{#4}
                \label{#5}
            \end{minipage}
        }
    \end{figure}
}

% #2 = caption
% #3 = label
% #4 = content
    \begin{table}[h]
        \centering
        \begin{tabular}{#1}
            #4
        \end{tabular}
        \caption{#2}
        \label{#3}
    \end{table}
}
